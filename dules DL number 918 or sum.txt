-- SPACE DOTS GIF EFFECT (Purple Cosmic Particles - Floating / Twinkling Stars)
local SpaceDotsContainer = Instance.new("Frame")
SpaceDotsContainer.Name = "SpaceDotsContainer"
SpaceDotsContainer.Size = UDim2.new(1, 0, 1, 0)
SpaceDotsContainer.BackgroundTransparency = 1
SpaceDotsContainer.Parent = mainFrame

local particles = {}
local particleCount = 32

for i = 1, particleCount do
    local p = Instance.new("Frame")
    p.Size = UDim2.new(0, math.random(2, 4), 0, math.random(2, 4))
    local hue = math.random(260, 300)
    p.BackgroundColor3 = Color3.fromHSV(hue/360, 0.85, 1)
    p.BackgroundTransparency = math.random(3, 7) / 10
    p.Position = UDim2.new(math.random(), 0, math.random(), 0)
    p.Parent = SpaceDotsContainer
   
    local pCorner = Instance.new("UICorner")
    pCorner.CornerRadius = UDim.new(1, 0)
    pCorner.Parent = p
   
    table.insert(particles, {
        obj = p,
        speed = math.random(8, 28) / 1000,
        sway = math.random(-12, 12) / 1000,
        alphaSpeed = math.random(40, 90) / 1000
    })
end

-- Animation Loop (Space Dots + Gradient Rotation)
game:GetService("RunService").RenderStepped:Connect(function(dt)
    StrokeGradient.Rotation = (StrokeGradient.Rotation + 0.8) % 360
   
    for _, pData in pairs(particles) do
        local p = pData.obj
        local newY = p.Position.Y.Scale + pData.speed
        local newX = p.Position.X.Scale + pData.sway * math.sin(tick() * 3)
       
        local trans = p.BackgroundTransparency
        trans = trans + pData.alphaSpeed
        if trans > 0.9 then trans = 0.1 end
        p.BackgroundTransparency = trans
       
        if newY > 1.1 then
            p.Position = UDim2.new(math.random(), 0, -0.1, 0)
        else
            p.Position = UDim2.new(newX, 0, newY, 0)
        end
    end
end)
