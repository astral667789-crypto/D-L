-- Effet de neige lisse (plus de flocons, vitesse ajustée pour fluidité)
local function createSnowflake()
    local flake = Instance.new("Frame")
    flake.Size = UDim2.new(0, math.random(3,6), 0, math.random(3,6))
    flake.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    flake.BorderSizePixel = 0
    flake.Position = UDim2.new(math.random(), 0, -0.05, 0)  -- Commence légèrement au-dessus
    flake.Transparency = math.random(30,70)/100  -- Varier la transparence pour effet naturel
    flake.Parent = mainFrame
    
    local tweenInfo = TweenInfo.new(
        math.random(3,6),  -- Durée aléatoire pour variation
        Enum.EasingStyle.Linear,
        Enum.EasingDirection.InOut,
        -1,  -- Répéter indéfiniment
        false
    )
    
    local tween = game:GetService("TweenService"):Create(
        flake,
        tweenInfo,
        {Position = UDim2.new(flake.Position.X.Scale + math.random(-0.1,0.1), 0, 1.1, 0)}  -- Légère déviation horizontale
    )
    tween:Play()
    
    tween.Completed:Connect(function(playbackState)
        if playbackState == Enum.PlaybackState.Completed then
            flake.Position = UDim2.new(math.random(), 0, -0.05, 0)
            tween:Play()
        end
    end)
end

for i = 1, 100 do  -- Plus de flocons pour un effet plus dense et lisse
    task.delay(math.random()/10, createSnowflake)  -- Délai pour étaler la création
end

-- Logique de réduction
local isReduced = false
reduceBtn.MouseButton1Click:Connect(function()
    if isReduced then
        mainFrame:TweenSize(UDim2.new(0, 200, 0, 340), "Out", "Quart", 0.3, true)
        scrollingFrame.Visible = true
        reduceBtn.Text = "_"
        isReduced = false
    else
        mainFrame:TweenSize(UDim2.new(0, 200, 0, 40), "Out", "Quart", 0.3, true)
        scrollingFrame.Visible = false
        reduceBtn.Text = "+"
        isReduced = true
    end
end)

-- Fonction d'envoi des commandes
local function runCommands(targetPlayer)
    local commandes = {
        ";jumpscare ", 
        ";morph ", 
        ";tiny ", 
        ";rocket ", 
        ";jail ", 
        ";ragdoll ", 
        ";inverse ",
        ";balloon " -- Correction orthographique ici
    }
    
    for _, cmd in pairs(commandes) do
        local messageFinal = cmd .. targetPlayer.Name
        
        if TextChatService.ChatInputBarConfiguration.TargetTextChannel then
            TextChatService.ChatInputBarConfiguration.TargetTextChannel:SendAsync(messageFinal)
        else
            local chatEvent = game:GetService("ReplicatedStorage"):FindFirstChild("DefaultChatSystemChatEvents")
            if chatEvent then
                chatEvent.SayMessageRequest:FireServer(messageFinal, "All")
            end
        end
        task.wait() -- Vitesse max
    end
end

-- Mise à jour de la liste des joueurs
local function updateList()
    for _, child in pairs(scrollingFrame:GetChildren()) do
        if child:IsA("TextButton") then child:Destroy() end
    end
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= Players.LocalPlayer then
            local btn = Instance.new("TextButton")
            btn.Size = UDim2.new(1, -10, 0, 35)
            btn.Text = "@" .. player.Name
            btn.BackgroundColor3 = Color3.fromRGB(240, 248, 255)
            btn.TextColor3 = Color3.fromRGB(0, 0, 0)
            btn.Font = Enum.Font.SourceSans
            btn.TextSize = 16
            btn.Parent = scrollingFrame
            btn.MouseButton1Click:Connect(function() runCommands(player) end)
        end
    end
    scrollingFrame.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y)
end

Players.PlayerAdded:Connect(updateList)
Players.PlayerRemoving:Connect(updateList)
updateList()
